2.1 The Tri-State Decision Boundary: when / then / end
The when statement is the primary decision gate for the AGI. Unlike an if statement, which checks a boolean condition, when is semantically coupled with the result<T> type and the TBB error sentinels. It is designed to safely "unwrap" reality. If a wave processor retrieves a pattern from the torus, that retrieval might fail due to destructive interference (returning ERR). The when statement forces the programmer to handle this dissonance before proceeding.
The audit of src/backend/ir/codegen_expr.cpp 1 confirms that while the type system knows about result<T> (mapped to opaque pointers), the structural logic to branch based on its content is missing.
2.1.1 Implementation Theory: The Branching Strategy
To implement visitWhenStmt, we must generate LLVM IR that performs a "validity check" rather than a simple truth check.
1. Condition Evaluation: We evaluate the expression.
2. Type Introspection: We check if the result is a result<T> struct or a TBB primitive.
3. Sentinel Detection:
   * If result<T>, we load the err field. If err is null, the state is Resonant (Success).
   * If tbb, we compare against the ERR sentinel (e.g., -128 for tbb8). If val!= ERR, the state is Resonant.
4. Scope Unwrapping: If Resonant, the value is extracted and bound to the loop variable (shadowing the raw result).
5. Branching: The control flow splits to the then block (Resonant path) or the end block (Dissonant path).
2.1.2 Concrete Implementation: visitWhenStmt
The following implementation fills the gap in codegen_stmt.cpp. It ensures that the AGI never operates on a "phantom" value derived from a failed memory lookup.

// src/backend/ir/codegen_stmt.cpp

/**
* @brief Generates LLVM IR for the 'when' tri-state logic gate.
* 
* Context: The Nikola AGI uses 'when' to filter high-confidence wave patterns.
* This implementation enforces safety by checking error sentinels before
* allowing access to the payload.
* 
* Logic:
* 1. Evaluate condition.
* 2. If Type is Result<T>: Check result.err == nullptr.
* 3. If Type is TBB: Check value!= TBB_SENTINEL.
* 4. Branch to 'then' or 'end'.
*/
llvm::Value* StmtCodegen::visitWhenStmt(WhenStmt* stmt) {
   // Establish the current function context for block insertion
   llvm::Function* function = builder.GetInsertBlock()->getParent();
   
   // Create basic blocks for the tri-state flow
   // "Then" handles the Resonant (success) state
   llvm::BasicBlock* thenBB = llvm::BasicBlock::Create(context, "when.resonant", function);
   // "End" handles the Dissonant (failure) state, if provided
   llvm::BasicBlock* endBB = stmt->elseBody? llvm::BasicBlock::Create(context, "when.dissonant", function) : nullptr;
   // "Merge" allows execution to rejoin
   llvm::BasicBlock* mergeBB = llvm::BasicBlock::Create(context, "when.converge", function);

   // Step 1: Generate the condition expression or declaration
   // The AGI often uses: when (val = torus.scan()) {... }
   llvm::Value* conditionValue = nullptr;
   
   if (stmt->isDeclaration) {
       // Codegen for 'var = expr', returns the value of expr
       conditionValue = exprCodegen->codegenDeclInExpr(stmt->declaration.get());
   } else {
       conditionValue = exprCodegen->codegen(stmt->condition.get());
   }

   if (!conditionValue) {
       throw std::runtime_error("Critical: Failed to generate 'when' condition signal.");
   }

   // Step 2: Determine validity (Resonance) based on type physics
   llvm::Type* condType = conditionValue->getType();
   llvm::Value* isResonant = nullptr;

   if (condType->isIntegerTy(1)) {
       // Standard Boolean: Direct mapping
       isResonant = conditionValue;
   } else if (exprCodegen->isResultType(condType)) {
       // Result<T>: The definition of Resonance is 'error is null'
       // Access index 0 (err field) of the Result struct
       llvm::Value* errPtrPtr = builder.CreateStructGEP(condType, conditionValue, 0, "res.err_ptr");
       llvm::Value* errPtr = builder.CreateLoad(llvm::PointerType::get(context, 0), errPtrPtr, "res.err");
       
       // Resonance Check: err == nullptr
       isResonant = builder.CreateIsNull(errPtr, "signal.is_pure");
   } else if (exprCodegen->isTBBType(condType)) {
       // TBB Integer: The definition of Resonance is 'value is not ERR'
       // Retrieve the specific sentinel for this width (e.g., -128 for tbb8)
       llvm::Value* errSentinel = exprCodegen->getTBBSentinel(condType);
       
       // Resonance Check: val!= ERR
       isResonant = builder.CreateICmpNE(conditionValue, errSentinel, "signal.is_valid");
   } else {
       // Safety Fallback: Treat as boolean (non-zero is true)
       // This handles standard integers used as flags
       llvm::Value* zero = llvm::Constant::getNullValue(condType);
       isResonant = builder.CreateICmpNE(conditionValue, zero, "signal.non_zero");
   }

   // Step 3: Conditional Branching
   // If resonant, enter the 'then' block. If dissonant, go to 'end' or 'merge'.
   llvm::BasicBlock* falseDest = endBB? endBB : mergeBB;
   builder.CreateCondBr(isResonant, thenBB, falseDest);

   // Step 4: Construct the 'Then' (Resonant) Block
   builder.SetInsertPoint(thenBB);
   
   // Safety: If the condition was a declaration (when x = val), 'x' is valid here.
   // The SymbolTable scope management handles this visibility.
   visitBlock(stmt->body.get());
   
   // Control Flow Integrity: Ensure the block terminates
   if (!thenBB->getTerminator()) {
       builder.CreateBr(mergeBB);
   }

   // Step 5: Construct the 'End' (Dissonant) Block
   if (endBB) {
       builder.SetInsertPoint(endBB);
       visitBlock(stmt->elseBody.get());
       if (!endBB->getTerminator()) {
           builder.CreateBr(mergeBB);
       }
   }

   // Step 6: Converge
   builder.SetInsertPoint(mergeBB);

   // Return void (Aria statements do not return values to the expression context)
   return llvm::Constant::getNullValue(llvm::Type::getInt32Ty(context));
}

2.2 Deterministic State Cascading: pick with fall
The pick statement is critical for the AGI's pattern matching. The torus creates complex interference patterns that must be categorized. A simple switch is insufficient because AGI states often overlap or require "cascading" logic (e.g., filtering a signal through multiple decay layers).
The user explicitly requested the fall statement implementation. In standard C++, break is required to stop fallthrough, which is dangerous. In Aria, fall is required to enable it.1 This inversion is a safety feature: accidental fallthrough is impossible. The fall(label) syntax adds a capability similar to a state machine transition, allowing the AGI to jump between non-adjacent case blocks, simulating "quantum tunneling" between logic states.
2.2.1 Implementation Theory: The Jump Table
The backend implementation utilizes LLVM's SwitchInst. The complexity lies in the fall statement. Since fall can target a specific label, we cannot simply rely on basic block ordering. We must build a map of Label -> BasicBlock during the initial pass of the pick statement and provide this map to the codegenFall visitor.
2.2.2 Concrete Implementation: visitPickStmt and codegenFall

// src/backend/ir/codegen_stmt.cpp

/**
* @brief Generates LLVM IR for the 'pick' pattern matching engine.
* 
* Context: Used by the AGI to categorize toroidal interference patterns.
* Supports explicit 'fall(label)' to simulate state transitions.
*/
llvm::Value* StmtCodegen::visitPickStmt(PickStmt* stmt) {
   // Step 1: Evaluate the Signal (Selector)
   llvm::Value* selector = exprCodegen->codegen(stmt->selector.get());
   if (!selector) return nullptr;

   llvm::Function* function = builder.GetInsertBlock()->getParent();
   llvm::BasicBlock* mergeBB = llvm::BasicBlock::Create(context, "pick.converge", function);
   llvm::BasicBlock* defaultBB = nullptr;

   // Step 2: Pre-allocate Logic Blocks (The Quantum States)
   // We create all blocks first to allow forward jumping via 'fall(label)'
   std::vector<llvm::BasicBlock*> caseBlocks;
   std::map<std::string, llvm::BasicBlock*> labeledBlocks;

   for (size_t i = 0; i < stmt->cases.size(); ++i) {
       std::string name = "pick.state." + std::to_string(i);
       llvm::BasicBlock* bb = llvm::BasicBlock::Create(context, name, function);
       caseBlocks.push_back(bb);
       
       // Register named states for direct transitions
       if (!stmt->cases[i]->label.empty()) {
           labeledBlocks[stmt->cases[i]->label] = bb;
       }
       
       // Identify the entropy sink (default case)
       if (stmt->cases[i]->isDefault) {
           defaultBB = bb;
       }
   }

   // Safety: Ensure a default block exists to catch unhandled entropy
   if (!defaultBB) {
       defaultBB = mergeBB;
   }

   // Step 3: Construct the Switch Vector (The Routing Table)
   // We use the size hint to optimize the lookup table generation
   llvm::SwitchInst* switchInst = builder.CreateSwitch(selector, defaultBB, stmt->cases.size());

   // Step 4: Populate the Routing Table and Generate Logic
   for (size_t i = 0; i < stmt->cases.size(); ++i) {
       PickCase* caseNode = stmt->cases[i].get();
       llvm::BasicBlock* currentBB = caseBlocks[i];
       
       // Add all matching constants to the switch instruction
       for (auto& valExpr : caseNode->values) {
           llvm::Value* val = exprCodegen->codegen(valExpr.get());
           if (llvm::ConstantInt* constVal = llvm::dyn_cast<llvm::ConstantInt>(val)) {
               switchInst->addCase(constVal, currentBB);
           } else {
               // Constraint: For v0.1.0, patterns must be compile-time constants
               // This ensures deterministic execution time for the physics loop
               throw std::runtime_error("Safety Violation: Non-constant pick case detected.");
           }
       }

       // Generate the logic within the state block
       builder.SetInsertPoint(currentBB);
       
       // Context Awareness: Push the 'fall' targets for this block
       // The 'next' block is simply i+1, but 'fall(label)' can target anything in labeledBlocks
       llvm::BasicBlock* nextSequentialBB = (i + 1 < caseBlocks.size())? caseBlocks[i+1] : defaultBB;
       
       // We push this context so 'codegenFall' knows where to jump
       pushFallContext(nextSequentialBB, labeledBlocks); 

       visitBlock(caseNode->body.get());

       popFallContext();

       // Control Flow Integrity: If the block didn't fall or return, it breaks to merge
       if (!currentBB->getTerminator()) {
           builder.CreateBr(mergeBB);
       }
   }

   // Step 5: Converge
   builder.SetInsertPoint(mergeBB);
   return nullptr;
}

/**
* @brief Generates IR for 'fall'.
* 
* Mechanism: Acts as a directed jump. Explicit fallthrough prevents 
* the "accidental casing" bugs common in C++.
*/
llvm::Value* StmtCodegen::codegenFall(FallStmt* stmt) {
   llvm::BasicBlock* target = nullptr;

   if (stmt->targetLabel.empty()) {
       // fall; -> Jumps to the immediate next physical block (cascade)
       target = getCurrentFallNext();
   } else {
       // fall(label); -> Jumps to a named state (tunneling)
       target = getLabeledBlock(stmt->targetLabel);
   }

   if (!target) {
       throw std::runtime_error("Logic Error: 'fall' target unreachable or undefined.");
   }

   builder.CreateBr(target);
   
   // Optimization: Create a dead block for any unreachable code following 'fall'
   // This satisfies LLVM's requirement that a block must have a terminator at the end
   llvm::Function* fn = builder.GetInsertBlock()->getParent();
   llvm::BasicBlock* deadBB = llvm::BasicBlock::Create(context, "after.fall", fn);
   builder.SetInsertPoint(deadBB);
   
   return nullptr;
}

2.3 Bounded Convergence: The till Statement
The till statement is conceptually distinct from while. A while loop is open-ended and potentially infinite—a dangerous property for an AGI that must respond to real-time inputs. The till loop implies convergence. It is used to iterate a process until a limit is reached (e.g., "iterate 100 times" or "till error < 0.001").1
The implementation provided below optimizes till into a canonical loop structure. This allows the LLVM backend to easily identify vectorization opportunities (SIMD), which is crucial for the performance of the 9D torus calculations.
2.3.1 Concrete Implementation: visitTillStmt

// src/backend/ir/codegen_stmt.cpp

/**
* @brief Generates LLVM IR for the 'till' bounded loop.
* 
* Syntax: till(limit, step) {... }
* Semantics: An implicit iterator '$' starts at 'limit' and approaches zero by 'step'.
* 
* Safety: Unlike 'while', 'till' guarantees a monotonic approach to termination 
* (assuming step is non-zero), preventing infinite stalls in the physics engine.
*/
llvm::Value* StmtCodegen::visitTillStmt(TillStmt* stmt) {
   llvm::Function* function = builder.GetInsertBlock()->getParent();
   llvm::BasicBlock* entryBB = builder.GetInsertBlock();
   llvm::BasicBlock* loopBB = llvm::BasicBlock::Create(context, "till.body", function);
   llvm::BasicBlock* exitBB = llvm::BasicBlock::Create(context, "till.exit", function);
   llvm::BasicBlock* condBB = llvm::BasicBlock::Create(context, "till.check", function);

   // Step 1: Evaluate Bounds
   // The AGI uses this for tensor dimension iteration (e.g., till(9, -1))
   llvm::Value* startVal = exprCodegen->codegen(stmt->limit.get());
   llvm::Value* stepVal = exprCodegen->codegen(stmt->step.get());
   
   // Validate step is not zero to prevent infinite loops (Safety)
   // In a production compiler, we might add a runtime check here.
   llvm::Value* zeroVal = llvm::ConstantInt::get(startVal->getType(), 0);

   // Step 2: Enter Loop Header
   builder.CreateBr(condBB);
   builder.SetInsertPoint(condBB);

   // Step 3: The Iterator ($) - PHI Node
   // The PHI node merges the value from the entry (start) and the back-edge (next)
   llvm::PHINode* iterator = builder.CreatePHI(startVal->getType(), 2, "iter.dollar");
   iterator->addIncoming(startVal, entryBB);

   // Step 4: Termination Check
   // Continue if iterator!= 0. 
   // This assumes the user provides a step that hits zero or passes it.
   // For strict safety, a signed check (iter > 0) might be preferred depending on step direction.
   llvm::Value* cond = builder.CreateICmpNE(iterator, zeroVal, "till.active");
   builder.CreateCondBr(cond, loopBB, exitBB);

   // Step 5: Loop Body
   builder.SetInsertPoint(loopBB);
   
   // Scope Management: Inject the '$' variable into the symbol table
   // This allows the AGI code to reference the current countdown value
   exprCodegen->setSymbol("$", iterator);
   
   visitBlock(stmt->body.get());

   // Step 6: The Latch (Update)
   // iterator = iterator + step
   llvm::Value* nextVal = builder.CreateAdd(iterator, stepVal, "till.next");
   iterator->addIncoming(nextVal, builder.GetInsertBlock());
   
   builder.CreateBr(condBB); // Jump back to header

   // Step 7: Exit
   builder.SetInsertPoint(exitBB);
   
   // Cleanup: Remove '$' from scope to prevent pollution
   exprCodegen->removeSymbol("$"); 

   return nullptr;
}

3. The Sentinel and the Void: Null and Error Safety Systems
In the 9D torus memory system, a "null" pointer is not just an empty value; it represents a topological gap in the memory manifold. If the AGI attempts to dereference a gap, it must not crash. It must either bridge the gap (Coalesce) or halt the local process safely (Unwrap/Panic).

3.2 The Null Coalesce (??) Operator
This operator provides a "default gravity" for the AGI. If a memory retrieval yields null or ERR, the ?? operator seamlessly substitutes a safe default value, preserving the continuity of the wave function.


C++




// src/backend/ir/codegen_expr.cpp (Addendum)

/**
* @brief Generates IR for 'lhs?? rhs'.
* 
* Safety: Implements short-circuit evaluation. If lhs is valid, rhs is NEVER executed.
* This is critical if rhs involves a costly or side-effecting operation (e.g., allocating memory).
*/
llvm::Value* ExprCodegen::visitNullCoalesceExpr(BinaryExpr* expr) {
   llvm::Function* function = builder.GetInsertBlock()->getParent();
   llvm::BasicBlock* lhsTrueBB = llvm::BasicBlock::Create(context, "nc.valid", function);
   llvm::BasicBlock* rhsBB = llvm::BasicBlock::Create(context, "nc.fallback", function);
   llvm::BasicBlock* mergeBB = llvm::BasicBlock::Create(context, "nc.merge", function);

   // 1. Evaluate LHS
   llvm::Value* lhsVal = codegen(expr->left.get());
   
   // 2. Check Validity (TBB-aware)
   // TBB types use specific bit patterns for ERR, not just 0.
   llvm::Value* isInvalid = nullptr;
   if (isTBBType(lhsVal->getType())) {
       llvm::Value* sentinel = getTBBSentinel(lhsVal->getType());
       isInvalid = builder.CreateICmpEQ(lhsVal, sentinel, "is_tbb_err");
   } else {
       // Standard pointers/optionals check for null
       isInvalid = builder.CreateIsNull(lhsVal, "is_null");
   }

   // 3. Branch
   builder.CreateCondBr(isInvalid, rhsBB, lhsTrueBB);

   // 4. LHS is Valid: Jump to merge
   builder.SetInsertPoint(lhsTrueBB);
   builder.CreateBr(mergeBB);

   // 5. LHS is Invalid: Evaluate RHS (The Fallback)
   builder.SetInsertPoint(rhsBB);
   llvm::Value* rhsVal = codegen(expr->right.get());
   builder.CreateBr(mergeBB);

   // 6. Merge Results via PHI Node
   builder.SetInsertPoint(mergeBB);
   llvm::PHINode* phi = builder.CreatePHI(lhsVal->getType(), 2, "nc.result");
   phi->addIncoming(lhsVal, lhsTrueBB);
   phi->addIncoming(rhsVal, rhsBB);

   return phi;
}

4. Appendage Theory: The Hybrid Memory Model
The "Nikola" AGI operates on a hybrid memory model described in the research as "Appendage Theory".1
* The Body (GC): High-level logic, cognitive processes, and strings live in Garbage Collected memory. This is safe but slow due to potential pauses.
* The Appendages (Wild): The 9D Torus physics engine runs in wild memory. This is raw, unmanaged, and extremely fast (SIMD-ready).
To make this work safely, the language requires specific operators to bridge the gap.
4.1 AddressOf (@) and Pinning (#)
The user noted that these were missing from the backend implementation.
* @var: Gets the raw address.
* #var: Pins a GC object. This is crucial. If the AGI passes a pointer to the physics engine, the GC must not move that object while the physics engine is calculating. The # operator creates a "Pin" scope.
4.1.1 Concrete Implementation: visitUnaryExpr


C++




// src/backend/ir/codegen_expr.cpp (Addendum)

llvm::Value* ExprCodegen::visitUnaryExpr(UnaryExpr* expr) {
   switch (expr->op) {
       case TokenType::TOKEN_AT: { // Address-Of (@var)
           // Implementation: Retrieve the ALLOCA (storage location), not the value.
           if (auto* idExpr = dynamic_cast<IdentifierExpr*>(expr->operand.get())) {
               llvm::Value* ptr = named_values[idExpr->name];
               if (!ptr) throw std::runtime_error("Unknown variable in address-of");
               return ptr; 
           }
           throw std::runtime_error("Operator '@' requires an l-value (variable).");
       }

       case TokenType::TOKEN_HASH: { // Pinning (#var)
           // Safety: This prevents the GC from moving the object during physics calculations.
           
           // 1. Get the object reference
           llvm::Value* objPtr = codegen(expr->operand.get());
           
           // 2. Call Runtime Pinning Intrinsic
           // void aria_gc_pin(void* obj);
           llvm::Function* pinFn = module->getFunction("aria_gc_pin");
           if (!pinFn) {
               // Declare intrinsic if missing
               llvm::Type* voidPtrTy = llvm::PointerType::get(context, 0);
               llvm::FunctionType* ft = llvm::FunctionType::get(
                   llvm::Type::getVoidTy(context), {voidPtrTy}, false);
               pinFn = llvm::Function::Create(ft, llvm::Function::ExternalLinkage, 
                                              "aria_gc_pin", module);
           }
           
           // Cast to opaque pointer and call pin
           llvm::Value* castObj = builder.CreatePointerCast(objPtr, builder.getPtrTy());
           builder.CreateCall(pinFn, { castObj });

           // 3. Return the pointer
           // The compiler's Borrow Checker (frontend) ensures this is eventually unpinned.
           return objPtr; 
       }
       
       //... standard unary ops (negate, not)
   }
   return nullptr;
}

5. The Self-Improvement Loop: JIT and wildx
The most novel and dangerous aspect of the Nikola AGI is its "self-improvement loop." It rewrites its own wave processing kernels. This requires Executable Memory (wildx). Standard allocators return Read/Write memory. Executing that memory causes a fault.
The user noted that wildx and the JIT integration were checking as incomplete.1
5.1 Safety Architecture: W^X (Write XOR Execute)
To protect the child companion from external attacks (e.g., prompt injection causing the AGI to execute malicious code), we cannot simply allocate RWX (Read-Write-Execute) memory. That is a massive security hole.
We implement a Staged JIT Allocator:
1. Allocation: Allocate memory as RW- (Read-Write). The AGI writes the new machine code here.
2. Sealing: Once written, the memory is "sealed" via aria_mem_protect_exec. This changes permissions to R-X (Read-Execute).
3. Execution: The AGI executes the kernel. It cannot be modified while running.
5.2 Concrete Runtime Implementation: jit_alloc.cpp
This file must be added to src/runtime/ to satisfy the linker for src/stdlib/jit.aria.


C++




// src/runtime/jit_alloc.cpp
// Implements the 'wildx' memory backing store with W^X protection.

#include <sys/mman.h> // POSIX memory management
#include <unistd.h>
#include <cstdint>
#include <errno.h>

extern "C" {

/**
* @brief Allocates raw executable-capable memory.
* Initial State: READ | WRITE (Not Executable yet)
* This prevents execution of incomplete code.
*/
void* aria_alloc_exec(int64_t size) {
   if (size <= 0) return nullptr;

   // 1. Align request to system page size (critical for mprotect)
   int64_t pageSize = sysconf(_SC_PAGESIZE);
   int64_t allocSize = (size + pageSize - 1) & ~(pageSize - 1);

   // 2. Map pages
   void* ptr = mmap(nullptr, allocSize, 
                    PROT_READ | PROT_WRITE, 
                    MAP_PRIVATE | MAP_ANONYMOUS, -1, 0);

   if (ptr == MAP_FAILED) return nullptr;
   return ptr;
}

/**
* @brief Seals the memory for execution.
* Transition: RW- -> R-X
* This makes the code immutable.
*/
int32_t aria_mem_protect_exec(void* ptr, int64_t size) {
   int64_t pageSize = sysconf(_SC_PAGESIZE);
   int64_t allocSize = (size + pageSize - 1) & ~(pageSize - 1);
   
   // Switch permissions. 
   // Attempting to write to this memory after this call will crash the AGI (Safety).
   return mprotect(ptr, allocSize, PROT_READ | PROT_EXEC);
}

/**
* @brief Frees JIT memory.
*/
void aria_free_exec(void* ptr, int64_t size) {
   int64_t pageSize = sysconf(_SC_PAGESIZE);
   int64_t allocSize = (size + pageSize - 1) & ~(pageSize - 1);
   munmap(ptr, allocSize);
}
