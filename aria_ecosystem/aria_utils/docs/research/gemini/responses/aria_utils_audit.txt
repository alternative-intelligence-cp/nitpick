/**
* @brief Generates Sentinel-Preserving Extension (SPE) code.
* 
* Logic:
* 1. Check if srcVal == SRC_SENTINEL.
* 2. Speculatively Sign-Extend srcVal to dstType (hardware efficient).
* 3. Select: If (1) is true, return DST_SENTINEL. Else return (2).
* 
* This compiles to branchless CMOV/CSEL instructions on x86/ARM, 
* ensuring no pipeline stalls in the AGI's tight loops.
*/
llvm::Value* TBBCodegen::generateWiden(llvm::Value* srcVal, Type* srcType, Type* dstType) {
   // 1. Identify Bit Widths
   unsigned srcWidth = getTBBBitWidth(srcType);
   unsigned dstWidth = getTBBBitWidth(dstType);
   
   // Safety Assertion: Widening only. Narrowing requires saturation logic (not covered here).
   if (srcWidth >= dstWidth) {
       // Logic error in compiler if this is reached
       return builder.CreateSExt(srcVal, builder.getIntNTy(dstWidth), "tbb.invalid_widen_fallback");
   }

   // 2. Generate APInt Sentinels (100...0 pattern)
   // using getSignedMinValue ensures we get the correct 0x80... form
   llvm::APInt srcSentinelInt = llvm::APInt::getSignedMinValue(srcWidth);
   llvm::APInt dstSentinelInt = llvm::APInt::getSignedMinValue(dstWidth);
   
   llvm::Value* srcSentinel = llvm::ConstantInt::get(context, srcSentinelInt);
   llvm::Value* dstSentinel = llvm::ConstantInt::get(context, dstSentinelInt);

   // 3. Emit The SPE Logic
   
   // A. Error Check: is_err = (src == src_sentinel)
   llvm::Value* isErr = builder.CreateICmpEQ(srcVal, srcSentinel, "tbb.is_err_widen");
   
   // B. Speculative Extension: extended = sext(src)
   // We perform this standard extension because it's required for the 'valid' path.
   llvm::IntegerType* dstLLVMType = builder.getIntNTy(dstWidth);
   llvm::Value* extendedVal = builder.CreateSExt(srcVal, dstLLVMType, "tbb.sext");
   
   // C. Sentinel Selection: result = select(is_err, dst_sentinel, extended)
   // If the input was an error, we force the output to be the wider error.
   // Otherwise, we carry over the value.
   llvm::Value* result = builder.CreateSelect(isErr, dstSentinel, extendedVal, "tbb.safe_widen");
   
   return result;
}

/**
* @brief Generates Sticky-Error Addition.
* 
* Logic:
* result = a + b
* if (a == ERR |

| b == ERR) result = ERR
* if (overflow(a + b)) result = ERR
* if (result == ERR) result = ERR (Sentinel Collision)
*/
llvm::Value* TBBCodegen::generateAdd(llvm::Value* lhs, llvm::Value* rhs, Type* type) {
   // 1. Get Sentinel for this type
   llvm::Value* sentinel = getErrSentinel(type);
   
   // 2. Input Validation (Sticky Property)
   // Check if either input is already broken.
   llvm::Value* lhsErr = builder.CreateICmpEQ(lhs, sentinel, "lhs.is_err");
   llvm::Value* rhsErr = builder.CreateICmpEQ(rhs, sentinel, "rhs.is_err");
   llvm::Value* anyInputErr = builder.CreateOr(lhsErr, rhsErr, "input.tainted");
   
   // 3. Arithmetic with Overflow Checking
   // Use LLVM intrinsic @llvm.sadd.with.overflow to catch wrapping.
   // TBB forbids wrapping; wrapping implies data corruption.
   llvm::Function* sadd = llvm::Intrinsic::getDeclaration(
       builder.GetInsertBlock()->getModule(), 
       llvm::Intrinsic::sadd_with_overflow, 
       { lhs->getType() }
   );
   
   llvm::Value* callRes = builder.CreateCall(sadd, {lhs, rhs}, "add.res");
   llvm::Value* mathRes = builder.CreateExtractValue(callRes, 0, "add.val");
   llvm::Value* overflow = builder.CreateExtractValue(callRes, 1, "add.ovf");
   
   // 4. Sentinel Collision Check
   // Corner case: A valid calculation (e.g., -64 + -64 in int8) might result in -128.
   // In TBB, -128 is reserved. So mathematically valid operations that land on the sentinel
   // must be treated as overflows to preserve the sentinel's meaning.
   llvm::Value* resIsSentinel = builder.CreateICmpEQ(mathRes, sentinel, "res.collision");
   
   // 5. Consolidate Failure Modes
   llvm::Value* calcFailed = builder.CreateOr(overflow, resIsSentinel, "calc.failed");
   llvm::Value* totalFailure = builder.CreateOr(anyInputErr, calcFailed, "tbb.failure");
   
   // 6. Final Selection
   return builder.CreateSelect(totalFailure, sentinel, mathRes, "tbb.add_result");
}

// Helper to determine bit width from Type* (assumes integration with Type system)
unsigned TBBCodegen::getTBBBitWidth(Type* type) {
   if (type->isTBB8()) return 8;
   if (type->isTBB16()) return 16;
   if (type->isTBB32()) return 32;
   if (type->isTBB64()) return 64;
   return 64; // Default/Fallback
}

llvm::Value* TBBCodegen::getErrSentinel(Type* type) {
   unsigned width = getTBBBitWidth(type);
   return llvm::ConstantInt::get(context, llvm::APInt::getSignedMinValue(width));
}

3. The Sensory Interface: Full Implementation of acurl
3.1 The Missing Link in Data Ingestion
The audit of the aria_utils source tree confirmed a critical gap: the src/acurl/ directory contains README specifications but no implementation code.1 acurl (Aria Curl) is designated as a Priority 1 utility. Its absence effectively blinds the AGI, preventing it from ingesting updates or communicating with distributed knowledge bases.
The core requirement for acurl is adherence to the Hex-Stream Topology. Standard curl writes both progress bars and data to stdout (or progress to stderr). This conflation is unacceptable for the AGI.
* FD 1 (stdout): User Interface (Progress bars, status).
* FD 3 (stddbg): Telemetry (Debug logs, JSON).
* FD 5 (stddato): Payload (The actual downloaded file).
3.2 The Deadlock Risk: Backpressure
A naïve implementation of acurl that simply writes downloaded bytes to a pipe (FD 5) introduces a deadly race condition known as Pipe Saturation Deadlock.
1. acurl downloads data faster than the AGI can process it.
2. The OS pipe buffer (typically 64KB) fills up.
3. acurl calls write() to the pipe.
4. write() blocks the thread, waiting for space.
5. Because the thread is blocked, libcurl stops sending "keep-alive" packets to the server.
6. The server times out the connection. Download fails.
To prevent this, we must use libcurl's CURL_WRITEFUNC_PAUSE feature. When the internal buffer is full, we tell libcurl to pause the network stream without closing the connection. We then drain the buffer to the pipe in a separate thread, and unpause libcurl only when space is available.2

// -----------------------------------------------------------------------------
// Thread-Safe Ring Buffer
// -----------------------------------------------------------------------------
class RingBuffer {
public:
   std::vector<char> data;
   size_t head = 0; // Write index
   size_t tail = 0; // Read index
   bool done = false; // Transfer complete flag
   std::mutex mtx;
   std::condition_variable cv_read;  // Wait for data
   std::condition_variable cv_write; // Wait for space

   RingBuffer(size_t size) : data(size) {}

   // Called by libcurl thread
   // Returns bytes written, or 0 if buffer full (signaling PAUSE)
   size_t write_data(char* ptr, size_t len) {
       std::unique_lock<std::mutex> lock(mtx);
       
       size_t capacity = data.size();
       size_t used = (head + capacity - tail) % capacity;
       size_t free_space = capacity - 1 - used;

       if (len > free_space) {
           // BACKPRESSURE: Buffer full. Signal libcurl to PAUSE.
           return 0; 
       }

       // Write data (handle wrap-around)
       size_t chunk1 = std::min(len, capacity - head);
       std::memcpy(&data[head], ptr, chunk1);
       if (chunk1 < len) {
           std::memcpy(&data, ptr + chunk1, len - chunk1);
       }
       
       head = (head + len) % capacity;
       
       lock.unlock();
       cv_read.notify_one(); // Wake up drainer
       return len;
   }

   // Called by drainer thread
   // Returns bytes read
   size_t read_to_fd(int fd) {
       std::unique_lock<std::mutex> lock(mtx);
       
       // Wait for data or completion
       cv_read.wait(lock, [this] { 
           return head!= tail |

| done; 
       });

       if (head == tail && done) return 0; // EOF

       size_t capacity = data.size();
       size_t contiguous = (head >= tail)? (head - tail) : (capacity - tail);
       
       // Write chunk to FD (BLOCKING WRITE to pipe)
       // We release the lock during write so ingestion doesn't block libcurl
       lock.unlock();
       
       ssize_t written = ::write(fd, &data[tail], contiguous);
       
       lock.lock();
       
       if (written > 0) {
           tail = (tail + written) % capacity;
           cv_write.notify_one(); // Wake up libcurl un-pauser
           return written;
       } else if (written < 0) {
           emit_telemetry("ERROR", "Write to pipe failed", errno);
           return 0; // Stop draining
       }
       return 0;
   }
   
   void finish() {
       std::lock_guard<std::mutex> lock(mtx);
       done = true;
       cv_read.notify_all();
   }
};

// Write Callback: Network -> Ring Buffer
static size_t write_callback(char *ptr, size_t size, size_t nmemb, void *userdata) {
   size_t bytes = size * nmemb;
   TransferContext* ctx = static_cast<TransferContext*>(userdata);
   
   size_t written = ctx->rb->write_data(ptr, bytes);
   
   if (written == 0 && bytes > 0) {
       // Buffer is full. Pause the transfer.
       emit_telemetry("WARN", "Buffer saturation. Pausing transfer.", 0);
       ctx->paused = true;
       return CURL_WRITEFUNC_PAUSE; 
   }
   
   return written;
}

           dprintf(FD_STDERR, "acurl error: %s\n", err.c_str());
           emit_telemetry("ERROR", err, res);
       }
       
       // Signal drainer to finish
       rb.finish();
       drainer.join();
       
       curl_easy_cleanup(ctx.curl);
   }

   curl_global_cleanup();
   emit_telemetry("INFO", "acurl finished", 0);
   return 0;
}

Architectural Analysis of the Solution:
1. Backpressure Valve: By checking free > BUFFER_SIZE / 2 in the progress callback, we implement a hysteresis loop. This prevents "thrashing" (rapid pause/unpause cycles) which consumes CPU and degrades network throughput.
2. Thread Safety: The implementation acknowledges libcurl's thread constraints. It strictly isolates the curl_easy_pause call to the main thread (inside the progress callback), triggered by state changes in the ring buffer.
3. Hex-Stream Compliance: The code explicitly directs output to the defined file descriptors, ensuring that stdout is purely UI. This allows acurl to be piped into atar without corrupting the archive stream.

5. Ecosystem Stability: Kernel-Level Stream Bootstrapping
5.1 The "Inheritance Gap"
The "Inheritance Gap" is a subtle but fatal bug identified in the ecosystem analysis.1 When the AGI is launched via a legacy shell script or systemd service, file descriptors 3, 4, and 5 are typically closed.
If an Aria utility starts and opens a file (e.g., a config file), the OS will assign it the lowest available FD—likely 3.
The utility then attempts to write telemetry to stddbg (FD 3).
Result: The utility overwrites its own configuration file with JSON logs. This is data corruption.
5.2 Remediation: aria_ensure_streams()
We cannot rely on the user to set up pipes correctly. Every Aria utility must self-heal its I/O environment at startup. This C++ routine checks the FDs and maps them to /dev/null if they are missing.
File: src/runtime/streams/bootstrap.cpp


C++




/**
* src/runtime/streams/bootstrap.cpp
* 
* I/O Topology Sanitizer.
* Prevents "The Inheritance Gap" data corruption.
*/

#include <unistd.h>
#include <fcntl.h>
#include <sys/stat.h>
#include <errno.h>

extern "C" void aria_init_streams() {
   // Defines the required topology: 0-5
   const int MAX_FD = 5;
   
   int null_fd = -1;

   for (int fd = 0; fd <= MAX_FD; fd++) {
       struct stat st;
       // fstat returns -1 and sets errno to EBADF if fd is closed
       if (fstat(fd, &st) == -1 && errno == EBADF) {
           
           // Lazy open of /dev/null only if needed
           if (null_fd == -1) {
               null_fd = open("/dev/null", O_RDWR);
               // If we can't open /dev/null, we are in deep trouble (chroot w/o devices?)
               // In AGI context, we must crash safely.
               if (null_fd == -1) _exit(127); 
           }

           // If the null_fd happened to grab the slot we wanted, great.
           if (null_fd == fd) {
               // The slot is now filled. We need a new 'master' null_fd for next iterations
               // because we can't use this one (it IS the target).
               // Actually, we can just dup it.
               null_fd = -1; // Force re-open next time or use dup
           } else {
               // Duplicate null_fd into the target slot
               if (dup2(null_fd, fd) == -1) _exit(126);
           }
       }
   }
   
   // Cleanup the master null_fd if it's dangling (i.e. not one of our target slots)
   if (null_fd > MAX_FD) close(null_fd);
}

This bootstrap routine must be linked into the crt0 (C Runtime Start) of every Aria executable or called explicitly at the top of main().
